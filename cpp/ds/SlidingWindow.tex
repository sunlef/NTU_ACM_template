有一个长为 $n$ 的序列 $a$，以及一个大小为 $k$ 的窗口。现在这个从左边开始向右滑动，每次滑动一个单位，求出每次滑动后窗口中的最大值和最小值。
\begin{lstlisting}
int n, k;
std::cin >> n >> k;

std::vector<int> a(n);
for (int &i : a) {
	std::cin >> i;
}

std::deque<int> q;
for (int i = 0; i < n; ++i) {  //输出最小值
	while (!q.empty() && a[q.back()] >= a[i]) {
		q.pop_back();
	}
	q.emplace_back(i);
	if (i >= k - 1) {
		while (!q.empty() && q.front() <= i - k) {
			q.pop_front();
		}
		std::cout << a[q.front()] << ' ';
	}
}
std::cout << '\n';

while (!q.empty()) {
	q.pop_back();
}

for (int i = 0; i < n; ++i) {  //输出最大值
	while (!q.empty() && a[q.back()] <= a[i]) {
		q.pop_back();
	}
	q.emplace_back(i);
	if (i >= k - 1) {
		while (!q.empty() && q.front() <= i - k) {
			q.pop_front();
		}
		std::cout << a[q.front()] << ' ';
	}
}
std::cout << '\n';
\end{lstlisting}
