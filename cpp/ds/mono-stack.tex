输出右边第一个比 $a_i$ 大的数的下标。
\begin{lstlisting}
int n;
std::cin >> n;

std::vector<int> a(n), ans(n);
for (int &i : a)
	std::cin >> i;

std::stack<int> s;
for (int i = n - 1; i >= 0; --i) {
	while (!s.empty() && a[s.top()] <= a[i]) 
		s.pop();
	if (s.empty()) 
		ans[i] = -1;
	else
		ans[i] = s.top();
	s.emplace(i);
}
\end{lstlisting}
