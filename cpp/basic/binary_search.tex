int binary_search_1(const std::vector<int> &a, int x) {
	// a 单调递增
	// 在 [0, n) 中查找 x 或 x 的后继的位置
	int n = a.size(), l = 0, r = n;
	while (l < r) {
		int mid = (l + r) / 2;
		if (a[mid] >= x) {
			r = mid;
		} else {
			l = mid + 1;
		}
	}
	return l;
}

int binary_search_2(const std::vector<int> &a, int x) {
	// a 单调递增
	//在 [0, n) 中查找 x 或 x 的前驱的位置
	int n = a.size(), l = 0, r = n;
	while (l < r) {
		int mid = (l + r + 1) / 2;
		if (a[mid] <= x) {
			l = mid;
		} else {
			r = mid - 1;
		}
	}
	return l;
}
